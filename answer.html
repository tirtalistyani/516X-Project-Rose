import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression
import statsmodels.api as sm

epth_df = depth_df = pd.read_excel(r"U:\downloads\project\Water Depth Data Bunny and Walnut 2016-2020 (1).xlsx",sheet_name = "Results")
precip_df = pd.read_csv(r"U:\downloads\project\PRISM_ppt_tmin_tmax_stable_4km_20160101_20201231_42.0783_-93.6311.csv")

depth_df = depth_df[['Location', 'Day','Result Value', 'Time']]
depth_df1 = depth_df[depth_df.Location == 'walnut_p1b']

depth_df ['Depth_max'] = depth_df["Result Value"]*1000
depth_df = depth_df.groupby(['Result Value','Day']).max()

import datetime
from datetime import datetime

precip_df['Day']= pd.to_datetime(precip_df['Day'])

merged_data = pd.merge(precip_df, depth_df, how="inner", on=['Day'])
y= merged_data['Depth_max']
x= merged_data['ppt (mm)']

model = LinearRegression(fit_intercept=True)
X = np.array([x]).reshape((-1, 1))
X.shape

model.fit(X,y)
print(model.intercept_)
print(model.coef_)

y_pred= model.predict(X)
y_pred

import matplotlib.pyplot as plt
import numpy as np

plt.rcdefaults()
fig, ax = plt.subplots()
plt.scatter (x,y, color = 'purple', s=100, alpha = 0.5)
plt.plot(X,y_pred, color = "black")
plt.title ('The relationship between precipitation and pothole water depth')
plt.xlabel ('Precipitation (mm)')
plt.ylabel ('Pothole water depth (mm)')
